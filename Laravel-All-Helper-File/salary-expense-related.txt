	 public function index(Request $request)
    {
        $data['salaryStatus'] = Salary::where('status',0)->get();

        if($request->startDate == "" && $request->endDate == "" && $request->input("year") == "" && $request->input("month") == "")
        {
            $EndDate =  date('Y-m-d');
            $start = date('Y-m-d');
            $end = date("Y-m-d",strtotime($EndDate."+1 day"));
            $data['salary'] = $data['salaryStatus']->whereBetween('created_at',[$start,$end]);
            $data['totalAmout'] = $data['salary']->sum('amount');
           return view('all-folder.admin.salary.show',$data);
        }
        elseif($request->startDate != "" && $request->endDate != "" && $request->input("year") == "" && $request->input("month") == "")
        {
            $startDate = $request->startDate;
            $endDate = $request->endDate;
            $data['startDate'] = $startDate;
            $data['EndDate'] =  $endDate;

            $EndDate =  $request->endDate;
            $start = date("Y-m-d",strtotime($request->input('startDate')));
            $end = date("Y-m-d",strtotime($EndDate."+1 day"));

            $data['totalAmout'] = $data['salary']->sum('amount');
           return view('all-folder.admin.salary.show',$data);
        }
        //------------------------------------------------
        if($request->startDate == "" && $request->endDate == "" && $request->input("year") != "" && $request->input("month") != "")

            $year = $request->input("year");

         return view('all-folder.admin.salary.show',$data);

	 public function create(Request $request)
    {
            
        if($request->get("month") == "" && $request->get("year") == "")
        {
            $monthCurrent = date('Y-m');

            $SearchSalaryCount =  Salary::where('date','like',"$monthCurrent%")->count();
            if($SearchSalaryCount > 0)
            {
                $SearchSalary =  Salary::where('date','like',"$monthCurrent%")->get();
        
                foreach( $SearchSalary as $searchId)
		{
                    $designation_id[] = $searchId->designation_id; 
                }
                $data['employee'] = Designation::where('role_id','!=', 11)
                                    ->WhereNotIn('id',$designation_id)
                                    ->get();
            }
            else{
                $data['employee'] = Designation::where('role_id','!=', 11)
                                    ->get();
            }

            $thismonth = null;
            Session::put('yearmonth', $thismonth);
                
            $data['advancedSalaryOfThisMonth'] = Salary_advance::where('date','like',"$monthCurrent%")->get();
            return view('all-folder.admin.salary.add',$data);    
        }
        else{
            $year = $request->input("year");
            $month = $request->input("month");
            $thismonths =  $year."-".$month;
            
            $thismonth = date('Y-m',strtotime($thismonths));
            $data['yearData'] = $year;
            $data['monthDataFull'] = date('F',strtotime($thismonths));
            $data['monthData'] = $month;
            Session::put('yearmonth',$thismonth);

            $searchSalaryCount = Salary::select('designation_id')->where('date','like',"$thismonth%")->count();
            if($searchSalaryCount > 0)
            {
                $SearchSalary =  Salary::where('date','like',"$thismonth%")->get();
    
                foreach( $SearchSalary as $searchId)
                {
                    $designation_id[] = $searchId->designation_id; 
                }
                $data['employee'] = Designation::where('role_id','!=', 11)
                                    ->WhereNotIn('id',$designation_id)
                                    ->get();
                $data['advancedSalaryOfThisMonth'] = Salary_advance::where('date','like',"$thismonth%")->get();
                return view('all-folder.admin.salary.add',$data);
            }
            $data['employee'] = Designation::where('role_id','!=',11)->get();
            $data['advancedSalaryOfThisMonth'] = Salary_advance::where('date','like',"$thismonth%")->get();
            return view('all-folder.admin.salary.add',$data);
        }
    }




 public function store(Request $request)
    {
        if($request->yearmonth == "")
        {
            $notification = array(
                'messege' => 'Please Select Month & Year First!',
                'alert-type' => 'error'
            );
            return redirect()->back()->with($notification);
        }


        $validator =  $this->validate($request, [
            'amount' => 'required|min:1|max:10',
            'bonus' => 'required|min:1|max:10',
            'penalty' => 'required|min:1|max:10',
            'designation_id' => 'required',
            'description' =>'nullable|max:150'
                ], 
                [
            'amount.required' => 'Amount field is required',
            'amount.min' => 'You may use Minimum 1 character',
            'amount.max' => 'You may use Maximum 10 character',

            'bonus.required' => 'Bonus field is required',
            'bonus.min' => 'You may use Minimum 1 character',
            'bonus.max' => 'You may use Maximum 10 character',

            'penalty.required' => 'Penalty field is required',
            'penalty.min' => 'You may use Minimum 1 character',
            'penalty.max' => 'You may use Maximum 10 character',

            'designation_id.required' => 'Employee Name field is required',

            'description.max' => 'You may use Maximum 150 character',
          
             ]   
            );

             $salary = new Salary(); 
             $salary->user_id = Auth::user()->id;
             $salary->designation_id = $request->designation_id;
             $salary->salary_advance_id = $request->salary_advance_id;
             $salary->date = $request->yearmonth."-01";
             $salary->amount = $request->amount;
             $salary->bonus = $request->bonus;
             $salary->penalty = $request->penalty;
            $save =  $salary->save();

            $thismonth = null;
            Session::put('yearmonth', $thismonth);

       if($save){
            $notification = array(
                'messege' => 'Salary is Paid  Successfully!',
                'alert-type' => 'success'
            );
            return redirect()->back()->with($notification);
        }else{
            $notification = array(
                'messege' => 'Salary is Not Paid!',
                'alert-type' => 'error'
            );
            return redirect()->back()->with($notification);
        }
    }

        $(document).ready(function () {
          
             $(".sal").click(function () {
                var id = parseInt($(this).attr("id").substr(4));

                var fix_salary = parseFloat($("#fixed_salary_" + id).text());
                var advance = parseFloat($("#advance_" + id).text());
                var payable = fix_salary - advance;
               if(isNaN(advance)){
                    $("#salary_" + id).text(fix_salary  + ' Tk.')
                    $("#salary_value_" + id).val(fix_salary)
                    $("#total_" + id).text(fix_salary);
                }else{
                    $("#salary_" + id).text(payable + ' TK.') 
                    $("#salary_value_" + id).val(payable) 
                    $("#total_" + id).text(payable);
                }
            });
            //--------------------------------------------
            $("body").on("change", ".slaryStatement", function () {
                //$('.slaryStatement').change(function () {
                var id = parseInt($(this).attr("id").substr(4));
               
                var salary = parseFloat($("#salary_" + id).text());
              
                var bonus = parseFloat($("#bon_" + id).val());
               
                var penalty = parseFloat($("#pen_" + id).val());
    
                var total_salary = salary + bonus - penalty;

                $("#total_" + id).text(total_salary);
    
            });
        });

    {

            'amount' => 'required|min:1|max:10',
            'employee_id' => 'required',
            'description' =>'nullable|max:150'
                ], 
                [
            'amount.required' => 'Amount field is required',
            'amount.min' => 'You may use Minimum 1 character',
            'amount.max' => 'You may use Maximum 10 character',

            'employee_id.required' => 'Employee Name field is required',

            'description.max' => 'You may use Maximum 150 character',
          
             ]   
            );

            $month          = date('Y-m');
            $amount         =  $request->amount;
            $employee_id    =  $request->employee_id;
            $description    =  $request->description;

        $isAdvancedPaied        =  Salary_advance::where('designation_id',$request->employee_id)->where('date','like',"$month%");
        $isAdvancedPaiedCount   =  $isAdvancedPaied->count();

            $notification = array(
                'messege' => 'You have already taken of this month !',

            );
            return redirect()->back()->with($notification);

            $salaryAdvance->designation_id = $request->employee_id;
            $salaryAdvance->date = date('Y-m-d');

            $salaryAdvance->description = $request->description;

                    'messege' => 'Salary Advanced is Not Paied!',

            }
        }
    }

 public function store(Request $request)
    {
        $validator =  $this->validate($request, [
            'cheque_no' => 'required|max:50|min:2',
            'bank_id' => 'required',
            'description' =>'nullable|max:150'
                ], 
                [
            'cheque_no.required' => 'Cheque No field is required',
            'cheque_no.min' => 'You may use Minimum 2 character',
            'cheque_no.max' => 'You may use Maximum 50 characte',

            'bank_id.required' => 'Bank field is required',

            'description.max' => 'You may use Maximum 150 character',
             ]   
            );

            $cheque = new Cheque();
            $cheque->user_id = Auth::user()->id;
            $cheque->cheque_no = $request->cheque_no;
            $cheque->bank_id = $request->bank_id;
            $cheque->description = $request->description;
            $save = $cheque->save();

            if($save){
                $notification = array(
                    'messege' => 'Cheque is Added Successfully!',
                    'alert-type' => 'success'
                );
                return redirect()->route('admin.cheque.index')->with($notification);
            }else{
                $notification = array(
                    'messege' => 'Cheque is Not Added!',
                    'alert-type' => 'error'
                );
                return redirect()->back()->with($notification);
            }
    }
